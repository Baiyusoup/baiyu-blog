## 深度优先遍历
```javascript
function dfs(node) {
  if (node === null) {
    return
  }
  console.log(node.val)
  dfs(node.left)
  dfs(node.right)
}

function dfs(node) {
  if (node === null) return;

  const stack = []
  stack.push(node);
  while(stack.length) {
    let p = stack.pop();
    console.log(p.val);
    if (node.left) stack.push(node.left)
    if (node.right) stack.push(node.right)
  }
}
```

## 广度优先遍历
```javascript
function bfs(node) {
  if (node === null) return;
  const queue = []
  queue.unshift(node);
  while(queue.length) {
    let p = queue.pop();
    console.log(p.val)
    if (node.left) queue.unshift(node.left)
    if (node.right) queue.unshift(node.right)
  }
}
```
