## 构建
```javascript
class Node {
  constructor(val, left = null, right = null) {
    this.val = val
    this.left = left
    this.right = right
  }
}
class BinaryTree {
  constructor() {
    this.head = null
  }
  insert(val) {
    if (this.head === null) {
      this.head = new Node(val)
    } else {
      this.insertNode(this.head, val)
    }
  }
  insertNode(tree, val) {
    if (tree.val < val) {
      if (tree.right === null) {
        tree.right = new Node(val)
      } else {
        this.insertNode(tree.right, val)
      }
    } else {
      if (tree.left === null) {
        tree.left = new Node(val)
      } else {
        this.insertNode(tree.left, val)
      }
    }
  }
}
```
